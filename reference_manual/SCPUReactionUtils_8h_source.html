48  scpu_data &data, scpu_data::entry_index idx1, scpu_data::entry_index idx2, scpu_data::new_entries &newEntries,
49  std::vector<scpu_data::entry_index> &decayedEntries, Reaction *reaction, const fix_pos &fixPos, reaction_record* record) {
50  auto& entry1 = data.entry_at(idx1);
51  auto& entry2 = data.entry_at(idx2);
52  if(record) {
53  record->type = static_cast<int>(reaction->getType());
54  record->where = (entry1.position() + entry2.position()) / 2.;
55  fixPos(record->where);
56  record->educts[0] = entry1.id;
57  record->educts[1] = entry2.id;
58  record->types_from[0] = entry1.type;
59  record->types_from[1] = entry2.type;
60  }
61  switch (reaction->getType()) {
62  case reaction_type::Decay: {
63  decayedEntries.push_back(idx1);
64  break;
65  }
66  case reaction_type::Conversion: {
67  entry1.type = reaction->getProducts()[0];
68  entry1.id = readdy::model::Particle::nextId();
69  if(record) record->products[0] = entry1.id;
70  break;
71  }
72  case reaction_type::Enzymatic: {
73  if (entry1.type == reaction->getEducts()[1]) {
74  // p1 is the catalyst
75  entry2.type = reaction->getProducts()[0];
76  entry2.id = readdy::model::Particle::nextId();
77  } else {
78  // p2 is the catalyst
79  entry1.type = reaction->getProducts()[0];
80  entry1.id = readdy::model::Particle::nextId();
81  }
82  if(record) {
83  record->products[0] = entry1.id;
84  record->products[1] = entry2.id;
85  }
86  break;
87  }
88  case reaction_type::Fission: {
89  auto n3 = readdy::model::rnd::normal3<readdy::scalar>(0, 1);
90  n3 /= std::sqrt(n3 * n3);
91 
92  readdy::model::Particle p(entry1.position() - reaction->getWeight2() * reaction->getProductDistance() * n3,
93  reaction->getProducts()[1]);
94  newEntries.emplace_back(p);
95 
96  entry1.type = reaction->getProducts()[0];
97  entry1.pos += reaction->getWeight1() * reaction->getProductDistance() * n3;
98  entry1.id = readdy::model::Particle::nextId();
99  fixPos(entry1.pos);
100  if(record) {
101  record->products[0] = entry1.id;
102  record->products[1] = p.getId();
103  }
104  break;
105  }
106  case reaction_type::Fusion: {
107  const auto e1Pos = data.entry_at(idx1).pos;
108  const auto e2Pos = data.entry_at(idx2).pos;
109  if (reaction->getEducts()[0] == entry1.type) {
110  entry1.pos += reaction->getWeight1() * (e2Pos - e1Pos);
111  } else {
112  entry1.pos += reaction->getWeight2() * (e2Pos - e1Pos);
113  }
114  fixPos(entry1.pos);
115  entry1.type = reaction->getProducts()[0];
116  entry1.id = readdy::model::Particle::nextId();
117  decayedEntries.push_back(idx2);
118  if(record) record->products[0] = entry1.id;
119  break;
120  }
121  }
122 }
