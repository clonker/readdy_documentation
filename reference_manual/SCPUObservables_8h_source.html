256 class SCPURadialDistribution : public readdy::model::observables::RadialDistribution {
257 public:
258  SCPURadialDistribution(KERNEL *const kernel, unsigned int stride, std::vector<scalar> binBorders, std::vector<std::string> typeCountFrom,
259  std::vector<std::string> typeCountTo, scalar particleToDensity) :
260  readdy::model::observables::RadialDistribution(kernel, stride, binBorders, typeCountFrom,
261  typeCountTo, particleToDensity), kernel(kernel) {}
262 
263  void evaluate() override {
264  if (binBorders.size() > 1) {
265  std::fill(counts.begin(), counts.end(), 0);
266  const auto particles = kernel->getKernelStateModel().getParticles();
267  auto isInCollection = [](const readdy::model::Particle &p, const std::vector<unsigned int> &collection) {
268  return std::find(collection.begin(), collection.end(), p.getType()) != collection.end();
269  };
270  const auto nFromParticles = std::count_if(particles.begin(), particles.end(),
271  [this, isInCollection](const readdy::model::Particle &p) {
272  return isInCollection(p, typeCountFrom);
273  });
274  {
275  const auto &distSquared = kernel->getKernelContext().getDistSquaredFun();
276  for (auto &&pFrom : particles) {
277  if (isInCollection(pFrom, typeCountFrom)) {
278  for (auto &&pTo : particles) {
279  if (isInCollection(pTo, typeCountTo) && pFrom.getId() != pTo.getId()) {
280  const auto dist = sqrt(distSquared(pFrom.getPos(), pTo.getPos()));
281  auto upperBound = std::upper_bound(binBorders.begin(), binBorders.end(), dist);
282  if (upperBound != binBorders.end()) {
283  const auto binBordersIdx = upperBound - binBorders.begin();
284  counts[binBordersIdx - 1]++;
285  }
286  }
287  }
288  }
289  }
290  }
291 
292  auto &radialDistribution = std::get<1>(result);
293  {
294  const auto &binCenters = std::get<0>(result);
295  auto &&it_centers = binCenters.begin();
296  auto &&it_distribution = radialDistribution.begin();
297  for (auto &&it_counts = counts.begin(); it_counts != counts.end(); ++it_counts) {
298  const auto idx = it_centers - binCenters.begin();
299  const auto lowerRadius = binBorders[idx];
300  const auto upperRadius = binBorders[idx + 1];
301  *it_distribution =
302  (*it_counts) /
303  (c_::four / c_::three * util::numeric::pi() * (std::pow(upperRadius, c_::three)
304  - std::pow(lowerRadius, c_::three))
305  * nFromParticles * particleToDensity);
306  ++it_distribution;
307  ++it_centers;
308  }
309  }
310  }
311  }
312 
313 protected:
314  KERNEL *const kernel;
315 };
