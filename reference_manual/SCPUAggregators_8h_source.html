51  void evaluate() override {
52  const auto &currentInput = std::get<0>(parentObservables)->getResult();
53  const auto &types = std::get<0>(currentInput);
54  const auto &ids = std::get<1>(currentInput);
55  const auto &positions = std::get<2>(currentInput);
56 
57  auto &resultTime = std::get<0>(result);
58  resultTime.push_back(getCurrentTimeStep());
59  auto &resultMsd = std::get<1>(result);
60  if (resultMsd.size() == 0) { // first evaluation
61  for (auto i = 0; i < positions.size(); ++i) {
62  if (typesToCount.size() == 0 || std::find(typesToCount.begin(), typesToCount.end(), types[i]) != typesToCount.end()) {
63  initialPositions.emplace(std::make_pair(ids[i], positions[i]));
64  }
65  }
66  resultMsd.push_back(0);
67  numberOfParticles.push_back(initialPositions.size());
68  return;
69  }
70  scalar msd = 0;
71  unsigned long currentNumberParticles = 0;
72  auto posIt = positions.begin();
73  auto idsIt = ids.begin();
74  for(; idsIt != ids.end(); ++idsIt, ++posIt) {
75  auto it = initialPositions.find(*idsIt);
76  if (it != initialPositions.end()) {
77  const auto displacement = *posIt - it->second;
78  msd += displacement * displacement;
79  ++currentNumberParticles;
80  }
81  }
82  if (currentNumberParticles > 0) {
83  msd /= static_cast<scalar>(currentNumberParticles);
84  };
85  resultMsd.push_back(msd);
86  numberOfParticles.push_back(currentNumberParticles);
87  }
88 
89 protected:
90  std::unordered_map<readdy::model::Particle::id_type, readdy::model::Vec3> initialPositions;
91  std::vector<unsigned long> numberOfParticles;
92  KERNEL *const kernel;
