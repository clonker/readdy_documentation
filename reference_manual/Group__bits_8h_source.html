81 inline std::vector<std::string> Group::sub_elements(H5O_type_t type) const {
82  std::vector<std::string> result;
83  auto group_info = info();
84  result.reserve(group_info.nlinks);
85  for(std::size_t i=0; i < group_info.nlinks; ++i) {
86  H5O_info_t oinfo;
87  H5Oget_info_by_idx(hid(), ".", H5_INDEX_NAME, H5_ITER_INC, i, &oinfo, H5P_DEFAULT);
88  if(oinfo.type == type) {
89  auto size = 1+ H5Lget_name_by_idx (hid(), ".", H5_INDEX_NAME, H5_ITER_INC, i, NULL, 0, H5P_DEFAULT);
90  if(size < 0) {
91  H5Eprint(H5Eget_current_stack(), stderr);
92  }
93 
94  std::vector<char> c_string (static_cast<std::size_t>(size));
95  H5Lget_name_by_idx (hid(), ".", H5_INDEX_NAME, H5_ITER_INC, i, c_string.data(), (std::size_t) size, H5P_DEFAULT);
96  std::string label (c_string.data());
97 
98  result.push_back(std::move(label));
99  }
100  }
101  return result;
102 }
