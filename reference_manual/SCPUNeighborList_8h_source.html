42 struct READDY_API ParticleIndexPair {
43  size_t idx1, idx2;
44 
45  ParticleIndexPair(size_t idx1, size_t idx2) {
46  if (idx1 < idx2) {
47  ParticleIndexPair::idx1 = idx1;
48  ParticleIndexPair::idx2 = idx2;
49  } else if (idx1 > idx2) {
50  ParticleIndexPair::idx1 = idx2;
51  ParticleIndexPair::idx2 = idx1;
52  } else {
53  throw std::runtime_error("pair must not have equal indices");
54  }
55  }
56 
57  friend size_t hash_value(const ParticleIndexPair &pip) {
58  size_t seed = 0;
59  readdy::util::hash::combine(seed, pip.idx1);
60  readdy::util::hash::combine(seed, pip.idx2);
61  return seed;
62  }
63 
64  friend bool operator==(const ParticleIndexPair &pip1, const ParticleIndexPair &pip2) {
65  return pip1.idx1 == pip2.idx1 && pip1.idx2 == pip2.idx2;
66  }
67 
68  friend std::ostream &operator<<(std::ostream &os, const ParticleIndexPair &pip) {
69  os << "ParticleIndexPair(" << pip.idx1 << ", " << pip.idx2 << ")";
70  return os;
71  }
72 };

181  virtual void setupBoxes(scalar skin) {
182  if (boxes.empty()) {
183  const auto simBoxSize = ctx->getBoxSize();
184  auto maxCutoff = ctx->calculateMaxCutoff();
185  if (maxCutoff > 0) {
186  maxCutoff += skin;
187  SCPUNotThatNaiveNeighborList::maxCutoff = maxCutoff;
188 
189  for (std::uint8_t i = 0; i < 3; ++i) {
190  nBoxes[i] = static_cast<int>(floor(simBoxSize[i] / maxCutoff));
191  if (nBoxes[i] == 0) nBoxes[i] = 1;
192  boxSize[i] = simBoxSize[i] / nBoxes[i];
193  }
194  for (long i = 0; i < nBoxes[0]; ++i) {
195  for (long j = 0; j < nBoxes[1]; ++j) {
196  for (long k = 0; k < nBoxes[2]; ++k) {
197  boxes.push_back({i, j, k, k + j * nBoxes[2] + i * nBoxes[2] * nBoxes[1]});
198  }
199  }
200  }
201  for (auto i = 0; i < nBoxes[0]; ++i) {
202  for (auto j = 0; j < nBoxes[1]; ++j) {
203  for (auto k = 0; k < nBoxes[2]; ++k) {
204  setupNeighboringBoxes(static_cast<unsigned long>(i),
205  static_cast<unsigned long>(j),

214  virtual void fillBoxes(const SCPUParticleData &data) {
215  const auto simBoxSize = ctx->getBoxSize();
216  if (maxCutoff > 0) {
217 
218  std::for_each(boxes.begin(), boxes.end(), [](Box& box) {box.particleIndices.clear();});
219  super::pairs->clear();
220 
221  unsigned long idx = 0;
222  const auto shift = readdy::model::Vec3(c_::half * simBoxSize[0], c_::half * simBoxSize[1],
223  c_::half * simBoxSize[2]);
224  for(const auto& entry : data) {
225  if(!entry.is_deactivated()) {
226  const auto pos_shifted = entry.position() + shift;
227  const long i = (const long) floor(pos_shifted[0] / boxSize[0]);
228  const long j = (const long) floor(pos_shifted[1] / boxSize[1]);
229  const long k = (const long) floor(pos_shifted[2] / boxSize[2]);
230  auto box = getBox(i, j, k);
231  if (box) {
232  box->particleIndices.push_back(idx);
233  }
234  }
235  ++idx;
236  }
237 
238  for (auto &&box : boxes) {
239  for (long i = 0; i < box.particleIndices.size(); ++i) {
240  const auto pI = box.particleIndices[i];
241  for (long j = i + 1; j < box.particleIndices.size(); ++j) {
242  super::pairs->push_back({pI, box.particleIndices[j]});
243  }
244 
245  for (auto &&neighboringBox : box.neighbors) {
246  for (const auto &pJ : neighboringBox->particleIndices) {
247  super::pairs->push_back({pI, pJ});
248  }
249  }
250  }
251  }
252  }
253  }

255  Box *getBox(long i, long j, long k) {
256  const auto &periodic = ctx->getPeriodicBoundary();
257  if (periodic[0]) i = readdy::util::numeric::positive_modulo(i, nBoxes[0]);
258  else if (i < 0 || i >= nBoxes[0]) return nullptr;
259  if (periodic[1]) j = readdy::util::numeric::positive_modulo(j, nBoxes[1]);
260  else if (j < 0 || j >= nBoxes[1]) return nullptr;
261  if (periodic[2]) k = readdy::util::numeric::positive_modulo(k, nBoxes[2]);
262  else if (k < 0 || k >= nBoxes[2]) return nullptr;
263  return &boxes[k + j * nBoxes[2] + i * nBoxes[2] * nBoxes[1]];
264  }
