109  DataSet createDataSet(const std::string &name, const std::vector<h5::h5_dims> &chunkSize,
110  const std::vector<h5::h5_dims> &maxDims, const DataSetType &memoryType,
111  const DataSetType &fileType, DataSetCompression compression = DataSetCompression::blosc) {
112  h5::h5_dims extensionDim;
113  {
114  std::stringstream result;
115  std::copy(maxDims.begin(), maxDims.end(), std::ostream_iterator<int>(result, ", "));
116  log::trace("creating data set with maxDims={}", result.str());
117  }
118  if (compression == DataSetCompression::blosc) {
119  blosc_compression::initialize();
120  }
121  // validate and find extension dim
122  {
123  auto unlimited_it = std::find(maxDims.begin(), maxDims.end(), h5::UNLIMITED_DIMS);
124  bool containsUnlimited = unlimited_it != maxDims.end();
125  if (!containsUnlimited) {
126  throw std::runtime_error("needs to contain unlimited_dims in some dimension to be extensible");
127  }
128  extensionDim = static_cast<h5::h5_dims>(std::distance(maxDims.begin(), unlimited_it));
129  log::trace("found extension dim {}", extensionDim);
130  }
131  h5::h5_handle handle;
132  {
133  // set up empty data set
134  std::vector<h5::h5_dims> dims(maxDims.begin(), maxDims.end());
135  dims[extensionDim] = 0;
136  DataSpace fileSpace(dims, maxDims);
137  DataSetCreatePropertyList propertyList;
138  propertyList.set_layout_chunked();
139  propertyList.set_chunk(chunkSize);
140  if (compression == DataSetCompression::blosc) propertyList.activate_blosc();
141 
142  auto _hid = H5Dcreate(hid(), name.c_str(), fileType.hid(),
143  fileSpace.hid(), H5P_DEFAULT, propertyList.hid(), H5P_DEFAULT);
144  if (_hid < 0) {
145  log::error("Error on creating data set {}", _hid);
146  H5Eprint(H5Eget_current_stack(), stderr);
147  throw std::runtime_error("Error on creating data set " + std::to_string(_hid));
148  } else {
149  handle = _hid;
150  }
151  }
152  DataSet ds(handle, memoryType, fileType);
153  ds.extensionDim() = extensionDim;
154  return ds;
155  }
